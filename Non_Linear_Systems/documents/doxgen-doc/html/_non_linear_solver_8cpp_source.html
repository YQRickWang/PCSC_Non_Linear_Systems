51 void NonLinearSolver::Bisection(double a, double b)
52 {
53  double tol = 1e-5;//tolerance of the method
54  double c = 0.0;
55  int k=0;
56  //first to check whether the dimension is 1
57  if(equations.GetDimension()>1)
58  {
59  std::cout<<"Bisection Method can only used for 1 dimension equation"<<std::endl;
60  return;
61  }
62 
63  if(equations.GetFunctionValue(a)*equations.GetFunctionValue(b)>=0){
64  std::cout<<"Incorrect a and b"<<std::endl;
65  return;
66  }
67 
68  while((b-a)>=tol){
69  c = (a+b)/2;
70  if(equations.GetFunctionValue(c)==0){
71  //c is the zero point
72  break;
73  }
74  else if(equations.GetFunctionValue(a)*equations.GetFunctionValue(c)<0.0){
75  b = c;
76  }
77  else{
78  a = c;
79  }
80  k++;
81  }
82 
83  //add to the zeropoint map
84  AddToZeroPoint("Bisection",c);
85 }

88 void NonLinearSolver::Aitken(double initial_guess, int max_iterations)
89 {
90  int dim = equations.GetDimension();
91  double tol = 1e-5;
92  double x_prev = initial_guess;
93  double x_next = 0.0;
94  double x_next_next = 0.0;
95  double error = 0.0;
96  int it_count = 0;
97 
98  if(dim>1)
99  {
100  std::cout<<"The Aitken is only for 1 dimension, please check again."<<std::endl;
101  return;
102  }
103 
104 
105  do{
106  x_next = equations.GetFpFunctionValue(x_prev);
107  x_next_next = equations.GetFpFunctionValue(x_next);
108 
109  //compute next by using aitken acceralation
110  x_next = x_prev - (pow(x_next-x_prev,2))/(x_next_next-2*x_next+x_prev);
111 
112  error = GetError(x_prev,x_next);
113  x_prev = x_next;
114  it_count++;
115  }while(error>=tol&&it_count<=max_iterations);
116 
117 
118  if(error>=tol)
119  std::cout<<"The aitken method does not converges."<<std::endl;
120 
121  //add to the zeroPoint
122  AddToZeroPoint("Aitken",x_next);
123 }
